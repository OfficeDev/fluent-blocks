import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs'

import { Illustration } from './StorybookIllustration'

  title="Media/Illustrations"
  component={Illustration}
  parameters={{ viewMode: 'docs' }}
  argTypes={{ ...contextArgTypes, illustration: { table: { disable: true } } }}

# Illustration

This library comes with a set of built-in illustrations you can simply reference by name in a `Figure` or `BigMessage` component.

      accentScheme: 'teams',
      illustration: 'default',
    }}
  >
